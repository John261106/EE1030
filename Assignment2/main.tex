\begin{enumerate}
	\item If $P(9a-2,-b)$ divides the line segment joining $A(3a+1,-3)$ and $B(8a,5)$  
 in the ratio $3:1$, find the values of $a$ and $b$.\\
 
 \solution As $P$ lies between $A$ and $B$, $P$ can be represented as \\ 
		$$P=\frac{kB+A}{k+1}$$ \\
		where $k$ is the ratio,here $k=3$\\
		$$P=\frac{3B+A}{4}=\frac{3\myvec{8a\\5}+\myvec{3a+1\\-3}}{4}=\frac{\myvec{27a+1\\12}}{4}$$
		also, $$P=\myvec{9a-2\\-b}$$\\
		on equating both sides and solving the two equations,we get \\
		$$a=1,b=-3$$











\end{enumerate}
